#include <iostream>
#include <iomanip>
#include <cmath>
#include <bitset>
using namespace std;

// int
int gcd(int a, int b){
    int start = 0;
    int end = 0;
    int ans = 0;

    if (a<b){
        start = b;
        end = a;
    }
    else if (b>a){
        start = a;
        end = b;
    }
    else{
        start = a;
        end = b;
    }
    for(int i = 1; i <= start; i++){
        if (start%i == 0 && end%i == 0){
            ans = i;
        }
    }
    return ans;
}


bool isPrime(int a){
    for (int i = 2; i < a; i++){
        if(a%i == 0 && a!=i){
            return false;
            break;
        }
    }
    return true;
}

int countPrimes(int a, int b){
    int prime = 0;
    for (int i = a; i <= b; i++){
        if(isPrime(i)==1 && i!=1){
            prime++;
        }
    }
    return prime;
}

// float
float sum_forward(int n){
    float sum = 0;
    for (float n = 1; n <= 100; n++){
        sum += 1/n;
    }
    return sum;
}

float sum_backward(int n){
    float sum = 0;
    for (float n = 100; n >= 1; n--){
        sum += 1/n;
    }
    return sum;
}

double hypot(double a, double b){
    float x = sqrt(pow(a, 2) + pow(b, 2));
    return x;
}

// array
float mean(float x[], int n){
    float sum = 0;
    for (int i = 0; i < n; i++){
        sum += float(x[i]);
    }
    return float(sum/n);
}

void double_each(int a[], int n){
    for (int i = 0; i < n; i++){
        a[i] *= 2;
    }
}

int prod(int x[], int n){
    int prod = 1;
    for (int i = 0; i < n; i++){
        prod *= x[i];
    }
    return prod;
}
